## Hook para Gerenciar QR Code

```typescript
// src/hooks/useQRCode.ts
import { useState, useCallback } from 'react';
import { Lote, Separacao, Produto } from '../types';
import toast from 'react-hot-toast';

interface QRData {
  tipo: 'LOTE' | 'SEPARACAO' | 'PRODUTO';
  id: string;
  [key: string]: any;
}

export function useQRCode() {
  const [lastScanned, setLastScanned] = useState<QRData | null>(null);
  const [scanHistory, setScanHistory] = useState<Array<{ data: QRData; timestamp: Date }>>([]);

  const processQRCode = useCallback((rawData: string | QRData): QRData | null => {
    try {
      // Se j√° √© objeto, usar direto
      if (typeof rawData === 'object') {
        return rawData;
      }

      // Tentar parsear JSON
      const data = JSON.parse(rawData);
      return data;
    } catch {
      // Se n√£o √© JSON, pode ser apenas n√∫mero do lote
      const numero = parseInt(rawData);
      if (!isNaN(numero)) {
        return { tipo: 'LOTE', id: '', numero };
      }
      return null;
    }
  }, []);

  const handleScan = useCallback((rawData: string | QRData) => {
    const data = processQRCode(rawData);
    
    if (!data) {
      toast.error('QR Code inv√°lido');
      return null;
    }

    setLastScanned(data);
    setScanHistory(prev => [
      { data, timestamp: new Date() },
      ...prev.slice(0, 9) // Manter √∫ltimos 10
    ]);

    return data;
  }, [processQRCode]);

  const findLoteByQR = useCallback((data: QRData): Lote | null => {
    // Buscar lote no localStorage
    const lotes = JSON.parse(localStorage.getItem('@tecnopano:lotes') || '[]');
    
    if (data.id) {
      return lotes.find((l: Lote) => l.id === data.id) || null;
    }
    if (data.numero) {
      return lotes.find((l: Lote) => l.numero === data.numero) || null;
    }
    return null;
  }, []);

  const findProdutoByQR = useCallback((data: QRData): Produto | null => {
    const produtos = JSON.parse(localStorage.getItem('@tecnopano:produtos') || '[]');
    
    if (data.id) {
      return produtos.find((p: Produto) => p.id === data.id) || null;
    }
    if (data.codigo) {
      return produtos.find((p: Produto) => p.codigo === data.codigo) || null;
    }
    return null;
  }, []);

  const generateQRData = useCallback((
    tipo: 'LOTE' | 'SEPARACAO' | 'PRODUTO',
    entity: Lote | Separacao | Produto
  ): QRData => {
    const base = { tipo, id: entity.id };

    switch (tipo) {
      case 'LOTE':
        const lote = entity as Lote;
        return {
          ...base,
          n: lote.numero,
          p: lote.totalKgNF,
          d: lote.dataEntrada,
        };
      case 'SEPARACAO':
        const sep = entity as Separacao;
        return {
          ...base,
          l: sep.loteId,
          t: sep.produtoTipoId,
          p: sep.pesoKg,
        };
      case 'PRODUTO':
        const prod = entity as Produto;
        return {
          ...base,
          c: prod.codigo,
          n: prod.nome,
        };
      default:
        return base;
    }
  }, []);

  return {
    lastScanned,
    scanHistory,
    handleScan,
    findLoteByQR,
    findProdutoByQR,
    generateQRData,
  };
}
```
